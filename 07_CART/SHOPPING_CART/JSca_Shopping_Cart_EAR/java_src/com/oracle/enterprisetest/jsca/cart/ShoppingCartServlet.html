070   public void doGet(HttpServletRequest request, HttpServletResponse response)

072

074     HttpSession session = null;
075     String serverName = null;

077     try {

079       out = response.getWriter();
080       // Set the content type in the response headers
081       response.setContentType("text/html");
082 
083       // print Header
084       printHeader(out);

086       // get a reference to the session object
087       session = request.getSession(true);
088 
089       serverName = MBeanHelper.getServerName();

091       // for load runner script validation
092       out.println("<h2>Servlet provided by: " + serverName + "</h2>");

094       // Get the "id" and "command" form paramters;
095       // if they're not available, the getParameter() method
096       // will return null.
097       String id = request.getParameter("id");
098       String command = request.getParameter("command");
099       if (command == null) {
100         command = "";
101       }
102 
103       // getRequestURI() is only one of many methods available in the
104       // HttpRequest object to inspect request properties.
105       String requestURI = request.getRequestURI();
106 
107       // get the customer's cart; if it the cart can't be retrieved
108       // from the HttpSession object, create a new cart.
109       ShoppingCart shopCart = (ShoppingCart) session.getAttribute("cart");
110       if (shopCart == null) {
111         shopCart = new ShoppingCart();
112         session.setAttribute("cart", shopCart);
113       }
114 
115       if (command.startsWith("add")) {
116         addItemToCart(shopCart, id);
117         session.setAttribute("cart", shopCart);
118         printCatalog{_DB_OPTION_}(out, requestURI);
119       } else if (command.equals("viewcart")) {
120         printCart{_DB_OPTION_}(shopCart, out, requestURI);
121       } else {
122         printCatalog{_DB_OPTION_}(out, requestURI);

126       out.println("<br>Current Session ID is :" + session.getId()
127           + "<br>");
128       out.close();
129     } catch (Exception e) {
130       // TODO Auto-generated catch block
131       e.printStackTrace();
132       if (out != null) {
133         out.println(getStackTrace(e));
134       }
135     } finally {
136       if (out != null) {
137         printFooter(out);
138       }
139     }

143   private void addItemToCart(ShoppingCart shopCart, String id) {
144     List<ShoppingCartItem> items = shopCart.getItems();
145     boolean found = false;
146     for (int i = 0; i < items.size(); i++) {
147       ShoppingCartItem shopItem = items.get(i);
148       String shopItemID = shopItem.getId() + "";
149       if (shopItemID.equals(id)) {
150         shopItem.setQuantity(shopItem.getQuantity() + 1);
151         found = true;
152       }
153     }
154     if (!found) {
155       ShoppingCartItem sItem = new ShoppingCartItem();
156       int sID = Integer.parseInt(id.trim());
157       sItem.setId(sID);
158       sItem.setQuantity(1);
159       sItem.setPrice(sID);
160       items.add(sItem);
161     }
162   }

185   /**
186    * Database access using JDBC is a straightforward operation; create a
187    * statement, execute it, retrieve the result set, and then iterate through
188    * it. One issue to watch out for with servlets is to free up the resources
189    * once you're done with them.
190    */
191   private void printCatalogFromDB(PrintWriter out, String requestURI)
192       throws Exception {
193     Statement stmt = null;
194     ResultSet rs = null;
195     Connection dbc = null;
196 
197     // Product list header
198     out.println("<h3>Product Catalog</h3>");
199     out.println("<form action=\"" + requestURI + "\" method=\"post\">");
200     out.println("<table border=1>");
201     out.println("<tr><th>Item</th><th>Price</th><th></th></tr>");
202 
203     try {
204       // setup jdbc statements
205       dbc = getConnection();
206       stmt = dbc.createStatement();
207       stmt.execute("select * from Products");
208       rs = stmt.getResultSet();
209 
210       // this is used format the money values
211       NumberFormat nf = NumberFormat.getCurrencyInstance();
212 
213       // iterate through the result set; outputing the product
214       // list in a html table
215       while (rs.next()) {
216         out.println("<tr>");
217         out.println("<td><b>" + rs.getString("name") + "</b><br>"
218             + rs.getString("description") + "</td>");
219         out.println("<td align=\"right\">"
220             + nf.format(rs.getDouble("price")) + "</td>");
221         out.println("<td><a href=\"" + requestURI + "?command=add&id="
222             + String.valueOf(rs.getInt("productID"))
223             + "\">Add to Cart</a></td>");
224         out.println("</tr>");
225       }
226       // Product list footer
227       out.println("</table>");
228       out.println("</form>");
229       out.println("<a href=\"" + requestURI
230           + "?command=viewcart\">View your Cart</a>");
231     } catch (Exception e) {
232       e.printStackTrace();
233       out.println("ERROR: " + e.toString());
234       throw e;
235     } finally {
236       closeResources(out, stmt, rs, dbc);
237     }
238   }

277   /**
278    * If the command "viewcart" is passed to the servlet, then this method is
279    * called. It iterates through the cart Hashtable, extracting the product
280    * details from the catalog. This method also calculates the individual item
281    * subtotals as well as the grand total of all the items in the customer's
282    * shopping cart.
283    */
284   private void printCartFromDB(ShoppingCart cart, PrintWriter out,
285       String requestURI) throws Exception {
286 
287     Statement stmt = null;
288     ResultSet rs = null;
289     Connection dbc = null;
290 
291     double totalPrice = getTotalPrice(cart);
292     int totalQuantity = getTotalQuantity(cart);
293     List<ShoppingCartItem> items = cart.getItems();
294 
295     NumberFormat nf = NumberFormat.getCurrencyInstance();
296 
297     out.println("<h1>Shopping Cart</h1>");
298     out.println("<h2>Your cart contains:</h2>");
299     out.println("<table border=1>");
300     out
301         .println("<tr><th>Quantity</th><th>Item</th><th>Price</th><th>Subtotal</th></tr>");
302     try {
303       dbc = getConnection();
304       for (int i = 0; i < items.size(); i++) {
305         ShoppingCartItem shopCartItem = items.get(i);
306         String id = shopCartItem.getId() + "";
307         Integer qty = (Integer) shopCartItem.getQuantity();
308         stmt = dbc.createStatement();
309         stmt.execute("select * from Products where productID = " + id);
310         rs = stmt.getResultSet();
311         while (rs.next()) {
312           double subtotal = rs.getDouble("price") * qty.intValue();
313           out.println("<tr>");
314           out.println("<td align=center>" + qty.toString() + "</td>");
315           out.println("<td>" + rs.getString("name") + "</td>");
316           out.println("<td>" + nf.format(rs.getDouble("price"))
317               + "</td>");
318           out.println("<td align=right>" + nf.format(subtotal)
319               + "</td>");
320           out.println("</tr>");
321         }
322         rs.close();
323         rs = null;
324         stmt.close();
325         stmt = null;
326       }
327       out
328           .println("<tr><td colspan=3 align=right>TOTAL Price:</td><td align=right>"
329               + nf.format(totalPrice) + "</td></tr>");
330       out
331           .println("<tr><td colspan=3 align=right>TOTAL Quantity:</td><td align=right>"
332               + totalQuantity + "</td></tr>");
333       out.println("</table>");
334       out.println("<p><a href=\"" + requestURI
335           + "\">Back to Shopping</a></p>");
336     } catch (Exception e) {
337       e.printStackTrace();
338       out.println("ERROR: " + e.toString());
339       throw e;
340     } finally {
341       closeResources(out, stmt, rs, dbc);
342     }
343   }
